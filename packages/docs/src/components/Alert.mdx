## 复杂一点，带有状态管理的使用

<Playground>
  {() => {
    const [status, setStatus] = React.useState('primary');
    function changeStatus() {
      setStatus(status === 'primary' ? 'error' : 'primary');
    }
    return (
      <div>
        <Alert type={status}>{status}</Alert>
        <Button onClick={changeStatus}>change status</Button>
      </div>
    )
  }}
</Playground>
